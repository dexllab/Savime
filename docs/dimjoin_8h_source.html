34 bool checkIntersection(TARPtr tar, SubtarPtr subtar1, SubtarPtr subtar2,
35  map<string, string> joinedDims,
36  map<string, JoinedRangePtr> &intersection,
37  StorageManagerPtr storageManager) {
38 #define IN_RANGE(X, Y, Z) ((X >= Y) && (X <= Z))
39 
40  for (auto entry : joinedDims) {
41  DimSpecPtr dimSpecs1 = subtar1->GetDimensionSpecificationFor(entry.first);
42  DimSpecPtr dimSpecs2 = subtar2->GetDimensionSpecificationFor(entry.second);
43  DimensionPtr dim =
44  tar->GetDataElement(LEFT_DATAELEMENT_PREFIX + entry.first)
45  ->GetDimension();
46  DimSpecPtr dimSpecs[2] = {dimSpecs1, dimSpecs2};
47  Literal lower[2], upper[2];
48  int64_t realLower[2], realUpper[2];
49 
50  for (int32_t i = 0; i < 2; i++) {
51  lower[i] = storageManager->Real2Logical(
52  dimSpecs[i]->dimension->GetDimension(), dimSpecs[i]->lower_bound);
53  upper[i] = storageManager->Real2Logical(
54  dimSpecs[i]->dimension->GetDimension(), dimSpecs[i]->upper_bound);
55  realLower[i] = storageManager->Logical2Real(dim, lower[i]);
56  realUpper[i] = storageManager->Logical2Real(dim, upper[i]);
57  }
58 
59  if (IN_RANGE(realLower[0], realLower[1], realUpper[1]) ||
60  IN_RANGE(realLower[1], realLower[0], realUpper[0])) {
61  JoinedRangePtr joinedRange = JoinedRangePtr(new JoinedRange());
62  joinedRange->lower_bound =
63  (realLower[0] > realLower[1]) ? realLower[0] : realLower[1];
64  joinedRange->upper_bound =
65  (realUpper[0] < realUpper[1]) ? realUpper[0] : realUpper[1];
66  intersection[dim->name] = joinedRange;
67  } else {
68  return false;
69  }
70  }
71 
72  return true;
73 }
