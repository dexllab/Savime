813 // Inserts a new data rectangle into the index structure.
814 // Recursively descends tree, propagates splits back up.
815 // Returns 0 if node was not split. Old node updated.
816 // If node was split, returns 1 and sets the pointer pointed to by
817 // new_node to point to the new node. Old node updated to become one of two.
818 // The level argument specifies the number of steps up from the leaf
819 // level to insert; e.g. a data rectangle goes in at level = 0.
820 RTREE_TEMPLATE
821 bool RTREE_QUAL::InsertRectRec(const Branch &a_branch, Node *a_node,
822  Node **a_newNode, int a_level) {
823  ASSERT(a_node && a_newNode);
824  ASSERT(a_level >= 0 && a_level <= a_node->m_level);
825 
826  // recurse until we reach the correct level for the new record. data records
827  // will always be called with a_level == 0 (leaf)
828  if (a_node->m_level > a_level) {
829  // Still above level for insertion, go down tree recursively
830  Node *otherNode;
831 
832  // find the optimal branch for this record
833  int index = PickBranch(&a_branch.m_rect, a_node);
834 
835  // recursively insert this record into the picked branch
836  bool childWasSplit = InsertRectRec(
837  a_branch, a_node->m_branch[index].m_child, &otherNode, a_level);
838 
839  if (!childWasSplit) {
840  // Child was not split. Merge the bounding box of the new record with the
841  // existing bounding box
842  a_node->m_branch[index].m_rect =
843  CombineRect(&a_branch.m_rect, &(a_node->m_branch[index].m_rect));
844  return false;
845  } else {
846  // Child was split. The old branches are now re-partitioned to two nodes
847  // so we have to re-calculate the bounding boxes of each node
848  a_node->m_branch[index].m_rect =
849  NodeCover(a_node->m_branch[index].m_child);
850  Branch branch(NUMDIMS);
851  branch.m_child = otherNode;
852  branch.m_rect = NodeCover(otherNode);
853 
854  // The old node is already a child of a_node. Now add the newly-created
855  // node to a_node as well. a_node might be split because of that.
856  return AddBranch(&branch, a_node, a_newNode);
857  }
858  } else if (a_node->m_level == a_level) {
859  // We have reached level for insertion. Add rect, split if necessary
860  return AddBranch(&a_branch, a_node, a_newNode);
861  } else {
862  // Should never occur
863  ASSERT(0);
864  return false;
865  }
866 }

869 // InsertRect provides for splitting the root;
870 // returns 1 if root was split, 0 if it was not.
871 // The level argument specifies the number of steps up from the leaf
872 // level to insert; e.g. a data rectangle goes in at level = 0.
873 // InsertRect2 does the recursion.
874 //
875 RTREE_TEMPLATE
876 bool RTREE_QUAL::InsertRect(const Branch &a_branch, Node **a_root,
877  int a_level) {
878  ASSERT(a_root);
879  ASSERT(a_level >= 0 && a_level <= (*a_root)->m_level);
880 #ifdef _DEBUG
881  for (int index = 0; index < NUMDIMS; ++index) {
882  ASSERT(a_branch.m_rect.m_min[index] <= a_branch.m_rect.m_max[index]);
883  }
884 #endif //_DEBUG
885 
886  Node *newNode;
887 
888  if (InsertRectRec(a_branch, *a_root, &newNode, a_level)) // Root split
889  {
890  // Grow tree taller and new root
891  Node *newRoot = AllocNode();
892  newRoot->m_level = (*a_root)->m_level + 1;
893 
894  Branch branch(NUMDIMS);
895

897  branch.m_rect = NodeCover(*a_root);
898  branch.m_child = *a_root;
899  AddBranch(&branch, newRoot, NULL);

903  branch.m_child = newNode;
904  AddBranch(&branch, newRoot, NULL);

910  }

912  return false;

967 // Pick a branch. Pick the one that will need the smallest increase
968 // in area to accomodate the new rectangle. This will result in the
969 // least total area for the covering rectangles in the current node.
970 // In case of a tie, pick the one which was smaller before, to get
971 // the best resolution when searching.
972 RTREE_TEMPLATE
973 int RTREE_QUAL::PickBranch(const Rect *a_rect, Node *a_node) {
974  ASSERT(a_rect && a_node);
975 
976  bool firstTime = true;
977  ELEMTYPEREAL increase;
978  ELEMTYPEREAL bestIncr = (ELEMTYPEREAL)-1;
979  ELEMTYPEREAL area;
980  ELEMTYPEREAL bestArea;
981  int best;
982  Rect tempRect(NUMDIMS);
983 
984  for (int index = 0; index < a_node->m_count; ++index) {
985  Rect *curRect = &a_node->m_branch[index].m_rect;
986  area = CalcRectVolume(curRect);
987  tempRect = CombineRect(a_rect, curRect);
988  increase = CalcRectVolume(&tempRect) - area;
989  if ((increase < bestIncr) || firstTime) {
990  best = index;
991  bestArea = area;
992  bestIncr = increase;
993  firstTime = false;
994  } else if ((increase == bestIncr) && (area < bestArea)) {
995  best = index;
996  bestArea = area;
997  bestIncr = increase;
998  }
999  }
1000  return best;
1001 }

1067 // The exact volume of the bounding sphere for the given Rect
1068 RTREE_TEMPLATE
1069 ELEMTYPEREAL RTREE_QUAL::RectSphericalVolume(Rect *a_rect) {
1070  ASSERT(a_rect);
1071 
1072  ELEMTYPEREAL sumOfSquares = (ELEMTYPEREAL)0;
1073  ELEMTYPEREAL radius;
1074 
1075  for (int index = 0; index < NUMDIMS; ++index) {
1076  ELEMTYPEREAL halfExtent = ((ELEMTYPEREAL)a_rect->m_max[index] -
1077  (ELEMTYPEREAL)a_rect->m_min[index]) *
1078  0.5f;
1079  sumOfSquares += halfExtent * halfExtent;
1080  }
1081 
1082  radius = (ELEMTYPEREAL)sqrt(sumOfSquares);
1083 
1084  // Pow maybe slow, so test for common dims like 2,3 and just use x*x, x*x*x.
1085  if (NUMDIMS == 3) {
1086  return (radius * radius * radius * m_unitSphereVolume);
1087  } else if (NUMDIMS == 2) {
1088  return (radius * radius * m_unitSphereVolume);
1089  } else {
1090  return (ELEMTYPEREAL)(pow(radius, NUMDIMS) * m_unitSphereVolume);
1091  }
1092 }

1129 // Method #0 for choosing a partition:
1130 // As the seeds for the two groups, pick the two rects that would waste the
1131 // most area if covered by a single rectangle, i.e. evidently the worst pair
1132 // to have in the same group.
1133 // Of the remaining, one at a time is chosen to be put in one of the two groups.
1134 // The one chosen is the one with the greatest difference in area expansion
1135 // depending on which group - the rect most strongly attracted to one group
1136 // and repelled from the other.
1137 // If one group gets too full (more would force other group to violate min
1138 // fill requirement) then other group gets the rest.
1139 // These last are the ones that can go in either group most easily.
1140 RTREE_TEMPLATE
1141 void RTREE_QUAL::ChoosePartition(PartitionVars *a_parVars, int a_minFill) {
1142  ASSERT(a_parVars);
1143 
1144  ELEMTYPEREAL biggestDiff;
1145  int group, chosen, betterGroup;
1146 
1147  InitParVars(a_parVars, a_parVars->m_branchCount, a_minFill);
1148  PickSeeds(a_parVars);
1149 
1150  while (
1151  ((a_parVars->m_count[0] + a_parVars->m_count[1]) < a_parVars->m_total) &&
1152  (a_parVars->m_count[0] < (a_parVars->m_total - a_parVars->m_minFill)) &&
1153  (a_parVars->m_count[1] < (a_parVars->m_total - a_parVars->m_minFill))) {
1154  biggestDiff = (ELEMTYPEREAL)-1;
1155  for (int index = 0; index < a_parVars->m_total; ++index) {
1156  if (PartitionVars::NOT_TAKEN == a_parVars->m_partition[index]) {
1157  Rect *curRect = &a_parVars->m_branchBuf[index].m_rect;
1158  Rect rect0 = CombineRect(curRect, &a_parVars->m_cover[0]);
1159  Rect rect1 = CombineRect(curRect, &a_parVars->m_cover[1]);
1160  ELEMTYPEREAL growth0 = CalcRectVolume(&rect0) - a_parVars->m_area[0];
1161  ELEMTYPEREAL growth1 = CalcRectVolume(&rect1) - a_parVars->m_area[1];
1162  ELEMTYPEREAL diff = growth1 - growth0;
1163  if (diff >= 0) {
1164  group = 0;
1165  } else {
1166  group = 1;
1167  diff = -diff;
1168  }
1169 
1170  if (diff > biggestDiff) {
1171  biggestDiff = diff;
1172  chosen = index;
1173  betterGroup = group;
1174  } else if ((diff == biggestDiff) && (a_parVars->m_count[group] <
1175  a_parVars->m_count[betterGroup])) {
1176  chosen = index;
1177  betterGroup = group;
1178  }
1179  }
1180  }
1181  Classify(chosen, betterGroup, a_parVars);
1182  }
1183 
1184  // If one group too full, put remaining rects in the other
1185  if ((a_parVars->m_count[0] + a_parVars->m_count[1]) < a_parVars->m_total) {
1186  if (a_parVars->m_count[0] >= a_parVars->m_total - a_parVars->m_minFill) {
1187  group = 1;
1188  } else {
1189  group = 0;
1190  }
1191  for (int index = 0; index < a_parVars->m_total; ++index) {
1192  if (PartitionVars::NOT_TAKEN == a_parVars->m_partition[index]) {
1193  Classify(index, group, a_parVars);
1194  }
1195  }
1196  }
1197 
1198  ASSERT((a_parVars->m_count[0] + a_parVars->m_count[1]) == a_parVars->m_total);
1199  ASSERT((a_parVars->m_count[0] >= a_parVars->m_minFill) &&
1200  (a_parVars->m_count[1] >= a_parVars->m_minFill));
1201 }
