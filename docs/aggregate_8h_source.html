69  void Configure() {
70  int32_t numDims = _dimensions.size();
71  for (int32_t i = 0; i < numDims; i++) {
72  auto dimName = _dimensions[i]->name;
73 
74  _multipliers[dimName] = 1;
75  _stride[dimName] = _dimensions[i]->GetLength();
76  _adj[dimName] = 1;
77 
78  for (int32_t j = i + 1; j < numDims; j++) {
79  int64_t preamble = _multipliers[dimName];
80  _multipliers[dimName] = preamble * _dimensions[j]->GetLength();
81  _stride[dimName] *= _dimensions[j]->GetLength();
82  _adj[dimName] *= _dimensions[j]->GetLength();
83  }
84  }
85 
86  if (numDims == 0) {
87  _stride["i"] = 1.0;
88  _adj["i"] = 1.0;
89  }
90  }
91 
92  int64_t GetTotalLength() {
93  int64_t totalLen = 1;
94 
95  for (DimensionPtr dim : _dimensions)
96  totalLen *= dim->GetLength();
97 
98  return totalLen;
99  }
