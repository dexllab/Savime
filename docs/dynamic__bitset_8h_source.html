256  template <typename BlockInputIterator>
257  void m_append(BlockInputIterator first, BlockInputIterator last,
258  std::forward_iterator_tag) {
259  assert(first != last);
260  block_width_type r = count_extra_bits();
261  std::size_t d = boost::detail::distance(first, last);
262  m_bits.reserve(num_blocks() + d);
263  if (r == 0) {
264  for (; first != last; ++first)
265  m_bits.push_back(*first); // could use vector<>::insert()
266  } else {
267  m_highest_block() |= (*first << r);
268  do {
269  Block b = *first >> (bits_per_block - r);
270  ++first;
271  m_bits.push_back(b | (first == last ? 0 : *first << r));
272  } while (first != last);
273  }
274  m_num_bits += bits_per_block * d;
275  }

548  class bit_appender {
549  // helper for stream >>
550  // Supplies to the lack of an efficient append at the less
551  // significant end: bits are actually appended "at left" but
552  // rearranged in the destructor. From the perspective of
553  // client code everything works *as if* dynamic_bitset<> had
554  // an append_at_right() function (eventually throwing the same
555  // exceptions as push_back) except that the function is in fact
556  // called bit_appender::do_append().
557  //
558  dynamic_bitset &bs;
559  size_type n;
560  Block mask;
561  Block *current;
562 
563  // not implemented
564  bit_appender(const bit_appender &);
565  bit_appender &operator=(const bit_appender &);
566 
567  public:
568  bit_appender(dynamic_bitset &r) : bs(r), n(0), mask(0), current(0) {}
569  ~bit_appender() {
570  // reverse the order of blocks, shift
571  // if needed, and then resize
572  //
573  std::reverse(bs.m_bits.begin(), bs.m_bits.end());
574  const block_width_type offs = bit_index(n);
575  if (offs)
576  bs >>= (bits_per_block - offs);
577  bs.resize(n); // doesn't enlarge, so can't throw
578  assert(bs.m_check_invariants());
579  }
580  inline void do_append(bool value) {
581 
582  if (mask == 0) {
583  bs.append(Block(0));
584  current = &bs.m_highest_block();
585  mask = Block(1) << (bits_per_block - 1);
586  }
587 
588  if (value)
589  *current |= mask;
590 
591  mask /= 2;
592  ++n;
593  }
594  size_type get_count() const { return n; }
595  };

774 template <typename Block, typename Allocator>
775 void dynamic_bitset<Block, Allocator>::resize(size_type num_bits,
776  bool value) // strong guarantee
777 {
778 
779  const size_type old_num_blocks = num_blocks();
780  const size_type required_blocks = calc_num_blocks(num_bits);
781 
782  const block_type v = value ? ~Block(0) : Block(0);
783 
784  if (required_blocks != old_num_blocks) {
785  m_bits.resize(required_blocks, v); // s.g. (copy)
786  }
787 
788  // At this point:
789  //
790  // - if the buffer was shrunk, we have nothing more to do,
791  // except a call to m_zero_unused_bits()
792  //
793  // - if it was enlarged, all the (used) bits in the new blocks have
794  // the correct value, but we have not yet touched those bits, if
795  // any, that were 'unused bits' before enlarging: if value == true,
796  // they must be set.
797 
798  if (value && (num_bits > m_num_bits)) {
799 
800  const block_width_type extra_bits = count_extra_bits();
801  if (extra_bits) {
802  assert(old_num_blocks >= 1 && old_num_blocks <= m_bits.size());
803 
804  // Set them.
805  m_bits[old_num_blocks - 1] |= (v << extra_bits);
806  }
807  }
808 
809  m_num_bits = num_bits;
810  m_zero_unused_bits();
811 }

898 //
899 // NOTE:
900 // Note that the 'if (r != 0)' is crucial to avoid undefined
901 // behavior when the left hand operand of >> isn't promoted to a
902 // wider type (because rs would be too large).
903 //
904 template <typename Block, typename Allocator>
905 dynamic_bitset<Block, Allocator> &dynamic_bitset<Block, Allocator>::
906 operator<<=(size_type n) {
907  if (n >= m_num_bits)
908  return reset();
909  // else
910  if (n > 0) {
911 
912  size_type const last = num_blocks() - 1; // num_blocks() is >= 1
913  size_type const div = n / bits_per_block; // div is <= last
914  block_width_type const r = bit_index(n);
915  block_type *const b = &m_bits[0];
916 
917  if (r != 0) {
918 
919  block_width_type const rs = bits_per_block - r;
920 
921  for (size_type i = last - div; i > 0; --i) {
922  b[i + div] = (b[i] << r) | (b[i - 1] >> rs);
923  }
924  b[div] = b[0] << r;
925 
926  } else {
927  for (size_type i = last - div; i > 0; --i) {
928  b[i + div] = b[i];
929  }
930  b[div] = b[0];
931  }
932 
933  // zero out div blocks at the less significant end
934  std::fill_n(m_bits.begin(), div, static_cast<block_type>(0));
935 
936  // zero out any 1 bit that flowed into the unused part
937  m_zero_unused_bits(); // thanks to Lester Gong
938  }
939 
940  return *this;
941 }

944 // NOTE:
945 // see the comments to operator <<=
946 //
947 template <typename B, typename A>
948 dynamic_bitset<B, A> &dynamic_bitset<B, A>::operator>>=(size_type n) {
949  if (n >= m_num_bits) {
950  return reset();
951  }
952  // else
953  if (n > 0) {
954 
955  size_type const last = num_blocks() - 1; // num_blocks() is >= 1
956  size_type const div = n / bits_per_block; // div is <= last
957  block_width_type const r = bit_index(n);
958  block_type *const b = &m_bits[0];
959 
960  if (r != 0) {
961 
962  block_width_type const ls = bits_per_block - r;
963 
964  for (size_type i = div; i < last; ++i) {
965  b[i - div] = (b[i] >> r) | (b[i + 1] << ls);
966  }
967  // r bits go to zero
968  b[last - div] = b[last] >> r;
969  }
970 
971  else {
972  for (size_type i = div; i <= last; ++i) {
973  b[i - div] = b[i];
974  }
975  // note the '<=': the last iteration 'absorbs'
976  // b[last-div] = b[last] >> 0;
977  }
978 
979  // div blocks are zero filled at the most significant end
980  std::fill_n(m_bits.begin() + (num_blocks() - div), div,
981  static_cast<block_type>(0));
982  }
983 
984  return *this;
985 }
